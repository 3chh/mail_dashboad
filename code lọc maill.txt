import os

import difflib
from email.header import decode_header
from datetime import datetime, timedelta

# --- 1. HÀM HỖ TRỢ HỆ THỐNG ---
def get_labels_from_excel(excel_path):
    label_map, warehouse_map = {}, {}
    if not os.path.exists(excel_path): return label_map, warehouse_map
    try:
        wb = load_workbook(excel_path, data_only=True)
        ws = wb.active
        for row in ws.iter_rows(min_row=2, values_only=True):
            mail_val = str(row[1]).lower().strip() if row[1] is not None else ""
            if mail_val: label_map[mail_val] = str(row[0]) if row[0] is not None else ""
            wh_name, addr_ref = row[3], row[4]
            if wh_name and addr_ref: warehouse_map[str(addr_ref)] = str(wh_name)
    except: pass
    return label_map, warehouse_map

def decode_mime(value):
    if not value: return ""
    try:
        parts = decode_header(value)
        return "".join([part.decode(enc or "utf-8", errors="ignore") if isinstance(part, bytes) else part for part, enc in parts])
    except: return str(value)

def get_body(message):
    body = ""
    try:
        if message.is_multipart():
            for part in message.walk():
                if part.get_content_type() in ["text/plain", "text/html"]:
                    payload = part.get_payload(decode=True)
                    if payload: body += payload.decode("utf-8", errors="ignore") + "\n"
        else:
            payload = message.get_payload(decode=True)
            if payload: body = payload.decode("utf-8", errors="ignore")
    except: pass
    return body

def extract_order_info(body, warehouse_map):
    tracking_val, address_val, warehouse_assigned = "N/A", "N/A", "N/A"
    track_match = re.search(r'【送り状番号】\s*(\d{12})', body)
    tracking_val = track_match.group(1) if track_match else (re.search(r'\d{12}', body).group(0) if re.search(r'\d{12}', body) else "N/A")
    lines = body.split('\n')
    for i, line in enumerate(lines):
        if "〒" in line:
            addr_part1 = line.strip()
            addr_part2 = (" " + lines[i+1].strip()) if (i+1 < len(lines) and lines[i+1].strip() and "【" not in lines[i+1]) else ""
            address_val = addr_part1 + addr_part2
            if warehouse_map:
                best_score, best_wh = -1, "N/A"
                clean_mail_addr = re.sub(r'\s+', '', address_val)
                for ref_addr, wh_name in warehouse_map.items():
                    score = difflib.SequenceMatcher(None, re.sub(r'\s+', '', ref_addr), clean_mail_addr).ratio()
                    if score > best_score: best_score, best_wh = score, wh_name
                warehouse_assigned = best_wh
            break
    return tracking_val, address_val, warehouse_assigned
